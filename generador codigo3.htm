                GENERAR CÓDIGO ✨

            </button>

        <!-- Estado de carga -->

                </div>

                <div class="border-t border-slate-800 pt-3">

                </div>

            </div>

        <!-- Error Log (Para saber qué falla) -->

        <div id="errorLog" class="hidden mt-4 p-3 bg-red-900/20 border border-red-500/50 rounded-lg text-[10px] text-red-400 font-mono"></div>

    </div>

 

    <script>

        const apiKey = ""; // El sistema inyecta la clave automáticamente

 

        async function ejecutarIA() {

            const input = document.getElementById('promptInput').value;

            if (!input.trim()) return;

 

            const btn = document.getElementById('mainBtn');

            const load = document.getElementById('loading');

            const res = document.getElementById('resultContainer');

            const errorDiv = document.getElementById('errorLog');

           

            // Reset de UI

            btn.disabled = true;

            btn.classList.add('opacity-50');

            load.classList.remove('hidden');

            res.classList.add('hidden');

            errorDiv.classList.add('hidden');

 

            try {

                const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

               

                const response = await fetch(url, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({

                        contents: [{

                            parts: [{

                                text: `Actúa como experto en MySQL. Genera SQL para trabajar con JSON: ${input}. Formato estricto: [SQL] tu_codigo [DESC] tu_explicacion`

                            }]

                        }]

                    })

                });

 

                if (!response.ok) {

                    const errorData = await response.json();

                    throw new Error(`HTTP ${response.status}: ${JSON.stringify(errorData)}`);

                }

 

                const data = await response.json();

                const text = data.candidates[0].content.parts[0].text;

 

                // Procesar respuesta

                const sqlMatch = text.split('[SQL]')[1]?.split('[DESC]')[0] || text;

                const descMatch = text.split('[DESC]')[1] || "Código generado con éxito.";

 

                document.getElementById('sqlOutput').textContent = sqlMatch.trim();

                document.getElementById('descOutput').textContent = descMatch.trim();

                res.classList.remove('hidden');

 

            } catch (err) {

                errorDiv.textContent = "ERROR TÉCNICO: " + err.message;

                errorDiv.classList.remove('hidden');

                console.error(err);

            } finally {

                btn.disabled = false;

                btn.classList.remove('opacity-50');

                load.classList.add('hidden');

            }

        }

 

        function copiarTexto() {

            const text = document.getElementById('sqlOutput').textContent;

            navigator.clipboard.writeText(text).then(() => {

                alert("¡Código copiado!");

            });

        }

    </script>
