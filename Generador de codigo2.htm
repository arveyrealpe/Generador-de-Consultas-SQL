        // API KEY MANAGEMENT

        const apiKey = "";

 

        async function fetchGemini(userPrompt) {

            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

           

            const systemPrompt = "Eres un experto en MySQL 8.0+. Genera código SQL para manejar datos JSON. IMPORTANTE: Estructura tu respuesta así: Primero el código SQL dentro de bloques ```sql y luego una explicación muy breve de máximo 2 líneas.";

           

            const payload = {

                contents: [{ parts: [{ text: userPrompt }] }],

                systemInstruction: { parts: [{ text: systemPrompt }] }

            };

 

            for (let i = 0; i < 5; i++) {

                try {

                    const response = await fetch(url, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    if (!response.ok) throw new Error('API Error');

                    const data = await response.json();

                    return data.candidates[0].content.parts[0].text;

                } catch (e) {

                    await new Promise(r => setTimeout(r, Math.pow(2, i) * 1000));

                }

            }

            throw new Error("No se pudo obtener respuesta tras varios intentos.");

        }

 

        async function generateSQL() {

            const input = document.getElementById('aiInput').value;

            if (!input.trim()) return;

 

            const btn = document.getElementById('generateBtn');

            const loading = document.getElementById('aiLoading');

            const container = document.getElementById('aiResponseContainer');

            const codeEl = document.getElementById('aiCodeResponse');

            const textEl = document.getElementById('aiTextResponse');

 

            btn.disabled = true;

            loading.classList.remove('hidden');

            container.classList.add('hidden');

 

            try {

                const response = await fetchGemini(input);

               

                // Extraer código de bloques markdown si existen

                const codeMatch = response.match(/```sql?([\s\S]*?)```/) || response.match(/```([\s\S]*?)```/);

                const cleanCode = codeMatch ? codeMatch[1].trim() : response.split('\n')[0];

                const cleanText = response.replace(/```[\s\S]*?```/g, '').trim();

 

                codeEl.textContent = cleanCode;

                textEl.textContent = cleanText || "Consulta generada basándose en tu solicitud.";

 

                container.classList.remove('hidden');

            } catch (err) {

                alert("Error: " + err.message);

            } finally {

                loading.classList.add('hidden');

                btn.disabled = false;

            }

        }

 

        function copyCode() {

            const text = document.getElementById('aiCodeResponse').textContent;

            const textArea = document.createElement("textarea");

            textArea.value = text;

            document.body.appendChild(textArea);

            textArea.select();

            document.execCommand('copy');

            document.body.removeChild(textArea);

           

            const toast = document.getElementById('toast');

            toast.style.opacity = "1";

            setTimeout(() => toast.style.opacity = "0", 2000);

        }
